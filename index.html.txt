
a=[int(i) for i in input().split()]
x=list.copy(a)
if len(a) == 1:
    print(a[0])
else:
    for i in range (len(a)):
        if i==0:
            x1=x[0]
            x[0]=a[1]+a[-1]
        elif i ==(len(a)-1):
            x[i] = a[i-1]+x1
        else:
            x[i] = a[i-1]+a[i+1]
        print(x[i])
